@()(implicit ats: ATS, contest: Contest)

@getRank(ranking: RankingData) = @{ranking.getRankFromZeroIn(ats.rankings())}

@getAwardLimit(ranking: RankingData) = @{ats.rankings().getAwardLimit(ranking.rule(contest))}

@line(ranking: RankingData) = @{
	val station = ats.stations().byCall(ranking.call).get(0)
	Seq(
		station.call,
		ranking.city,
		ranking.sect,
		if(getRank(ranking) == 0) "優勝" else s"${1+getRank(ranking)}位",
		if(getRank(ranking) < getAwardLimit(ranking)) "入賞局" else "",
		ranking.total,
		station.name,
		station.post,
		station.addr,
		station.mail,
		station.note
	).mkString(",").concat("\n")
}
