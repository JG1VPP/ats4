@()(implicit db: Database)

@import models.Call

@line(team: Team, game: Game) = @{
	Seq(
		team.call,
		game.city,
		game.sect,
		if(game.rank == 0) "優勝" else "%d位".format(1+game.rank),
		if(game.best) "入賞局" else "",
		game.total,
		team.name,
		team.addr,
		team.mail,
		team.comm
	).mkString(",").concat("\n")
}

@for(sect <- Sections.all; (team,game) <- Sect(sect.getName).game.sortBy(-_.total).map(m=>Call(m.call).team.head->m)) {@line(team,game)}
