@line(ranking: RankingData) = @{
	val station = StationData.findAllByCall(ranking.call).head
	Seq(
		station.call,
		ranking.city,
		ranking.sect,
		if(Rank.rank(ranking) == 0) "優勝" else "%d位".format(1+Rank.rank(ranking)),
		if(Rank.glad(ranking)) "入賞局" else "",
		ranking.rate,
		station.name,
		station.post,
		station.addr,
		station.mail,
		station.note
	).mkString(",").concat("\n")
}
