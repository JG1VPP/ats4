@line(record: Record) = @{
	val person = Person.findAllByCall(record.call).head
	Seq(
		person.call,
		record.city,
		record.sect,
		if(Rank.rank(record) == 0) "優勝" else "%d位".format(1+Rank.rank(record)),
		if(Rank.glad(record)) "入賞局" else "",
		record.total,
		person.name,
		person.addr,
		person.mail,
		person.comm
	).mkString(",").concat("\n")
}
